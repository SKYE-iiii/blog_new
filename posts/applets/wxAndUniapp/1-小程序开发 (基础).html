                            # 六. 小程序生命周期

                            1. onLaunch : 小程序初始化时执行 , 全局只执行一次
                            2. onShow : 小程序显示.
                            3. onHide : 小程序隐藏. 页面从前台变为后台时执行

                            # 七. 页面生命周期

                            1. onLoad : 页面创建时执行 (只执行一次)
                            2. onShow : 页面出现在前台时执行
                            3. onReady : 页面首次渲染完毕时执行 (只执行一次)
                            4. onHide : 页面从前台变为后台时执行
                            5. onUnload : 页面销毁时执行 (只执行一次)
                              1. ps : onUnload 一般不会执行 , 除非特殊场景; 如 : 使用 wx.reLaunch , wx.navigateBack, wx.redirectTo 跳转当前页面,则前页面的生命周期会执行 onUnload 函数

                            接口调用报错 : https://geoapi.qweather.comopen in new window 不在以下 request 合法域名列表中，请参考文档：https://developers.weixin.qq.com/miniprogram/dev/framework/ability/network.htmlopen in new window 解决方案 :

                            1. 开发阶段 : 详情 -> 本地设置 -> 勾选不校验合法域名
                            2. 上线阶段 : 绑定请求接口域名 ; 步骤 : 登录微信公众平台 -> 开发 -> 开发管理 -> 开发设置 -> 服务器域名 -> request 合法域名 : https://devapi.qweather.comopen in new window
                            1. 封装 API 请求
                              1. utils/api.js
                            const BASE_URL = "https://devapi.qweather.com/v7";
const KEY = "652273920efd481f81c715a1d285fbfa";
/**
 * API请求函数
 */
const request = (url, method, data) => {
  // 设置请求 KEY
  data.key = KEY;
  return new Promise((resolve, reject) => {
    wx.request({
      url: url,
      method,
      data,
      header: {
        "content-type": "application/json", // 默认值
      },
      success(res) {
        resolve(res.data);
      },
      fail(err) {
        reject(err);
      },
    });
  });
};
/**
 * 导出接口
 */
module.exports = {
  threeDays: (data) => {
    return request(BASE_URL + "/weather/3d", "get", data);
  },
};
